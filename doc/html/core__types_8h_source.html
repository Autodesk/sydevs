322 inline std::string core_type<arraynd<T, ndims>>::tostring(const arraynd<T, ndims>& X)
323 {
324  auto indices = std::array<int64, ndims>();
325  for (int64 idim = 0; idim < ndims; ++idim) {
326  indices[idim] = 0;
327  }
328  int64 offset = X.offset();
329  int64 idim = 0;
330  auto s = std::string("{");
331  while (idim >= 0) {
332  if (indices[idim] < X.dims()[idim]) {
333  if (indices[idim] > 0) {
334  s += ", ";
335  }
336  if (idim == ndims - 1) {
337  s += core_type<T>::tostring(X.data()[offset]);
338  ++indices[idim];
339  offset += X.strides()[idim];
340  }
341  else {
342  ++idim;
343  s += "{";
344  }
345  }
346  else {
347  indices[idim] = 0;
348  offset -= X.dims()[idim]*X.strides()[idim];
349  --idim;
350  if (idim >= 0) {
351  ++indices[idim];
352  offset += X.strides()[idim];
353  s += "}";
354  }
355  }
356  }
357  s += "}";
