373 inline duration collection_node<AgentID, Node>::handle_planned_event(duration elapsed_dt)
374 {
375  auto& current_t = event_time().t();

380  if (agent_index < macro_event_id) {
381  // The planned event is triggered by an agent.
382  // Execute an agent planned event followed by a micro planned event.
383  auto& agent = handle_agent_planned_event(agent_index);
384  int64 list_size = agent_IO(agent).message_output_list_size();
385  const auto& agent_id = agent_ids_[agent_index];

387  for (int64 list_index = 0; list_index < list_size; ++list_index) {

390  const auto& val = agent_IO(agent).message_output_value(list_index);
391  external_IO().print_event("micro-planned");
392  external_IO().print_elapsed_duration(micro_elapsed_dt);
393  prototype_IO().append_message_output(port_index, val);
394  auto dt = micro_planned_event(agent_id, micro_elapsed_dt);
395  prototype_IO().clear_flow_inputs();
396  prototype_IO().clear_message_input();
397  prototype_IO().clear_message_outputs();
398  prototype_IO().clear_flow_outputs();
399  auto macro_planned_dt = scale_planned_dt(dt);
400  external_IO().print_planned_duration(macro_planned_dt);
401  micro_elapsed_dt -= micro_elapsed_dt;
402  if (macro_planned_dt.finite()) {
403  t_queue_.plan_event(macro_event_id, macro_planned_dt);
404  }
405  else {
406  t_queue_.cancel_event(macro_event_id);
407  }
408  if (time_precision() != no_scale) {

410  }

414  else {

419  auto dt = macro_planned_event(elapsed_dt);

427  }

430  }
431  if (time_precision() != no_scale) {

433  }

677  agent_IO(agent).print_flow_input(port_index);
678  }
679  agent_IO(agent).activate(flow, input);
680  agent.process_initialization_event();
681  agent_IO(agent).deactivate();

686  }

688  for (int64 port_index = 0; port_index < output_port_count; ++port_index) {
689  agent_IO(agent).print_flow_output(port_index);
690  const auto& flow_output_val = agent_IO(agent).flow_output_port_value(port_index);
691  prototype_IO().assign_flow_output(port_index, flow_output_val);
692  }

770 template<typename AgentID, typename Node>
771 inline Node& collection_node<AgentID, Node>::handle_agent_planned_event(int64 agent_index)
772 {
773  event_time().advance();
774  auto& agent = *dynamic_cast<Node*>(&internal_structure().node(agent_index));
775  agent_IO(agent).print_event("planned");
776  auto elapsed_dt = duration();
777  if (agent.time_precision() != no_scale) {
778  elapsed_dt = t_cache_.duration_since(agent_index).fixed_at(agent.time_precision());
779  }
780  agent_IO(agent).activate(message, output);
781  auto planned_dt = agent.process_planned_event(elapsed_dt);
782  agent_IO(agent).deactivate();
783  if (planned_dt.finite()) {
784  t_queue_.plan_event(agent_index, planned_dt);
785  }
786  else {
787  t_queue_.pop_imminent_event(agent_index);
788  }
789  if (agent.time_precision() != no_scale) {
790  t_cache_.retain_event(agent_index, agent.time_precision());
791  }
792  int64 list_size = agent_IO(agent).message_output_list_size();
793  for (int64 list_index = 0; list_index < list_size; ++list_index) {
794  int64 port_index = agent_IO(agent).message_output_index(list_index);
795  agent_IO(agent).print_message_output(list_index, port_index);
796  }
797  return agent;
798 }
